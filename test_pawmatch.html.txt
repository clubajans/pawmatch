        // Google Giriş (Samsun Ağı için Yönlendirme Metodu)
        function googleLogin() {
            const provider = new firebase.auth.GoogleAuthProvider();
            auth.signInWithRedirect(provider);
        }

        // Giriş Durumunu İzle
        auth.onAuthStateChanged((user) => {
            if (user) {
                document.getElementById('login-area').style.display = 'none';
                document.getElementById('user-info').style.display = 'block';
                document.getElementById('welcome-msg').innerText = "Merhaba, " + user.displayName;
                console.log("Giriş yapıldı:", user.uid);
            } else {
                document.getElementById('login-area').style.display = 'block';
                document.getElementById('user-info').style.display = 'none';
            }
        });

        // Sayfa yönlendirmesini kontrol et
        auth.getRedirectResult().then(result => {
            if (result && result.user) {
                alert("Samsun PawMatch Ağına Hoş Geldin " + result.user.displayName);
            }
        });

        function loadLost() {
            db.collection("lostPets").orderBy("createdAt", "desc").onSnapshot(snap => {
                const list = document.getElementById('lost-list');
                list.innerHTML = "";
                snap.forEach(doc => {
                    const d = doc.data();
                    list.innerHTML += `<div class="item-card"><div class="item-body"><b>${d.location}</b><p>${d.description || ''}</p><a href="tel:${d.phone}" class="action-btn" style="background:var(--secondary)">ARA</a></div></div>`;
                });
            });
        }

        function loadAdaylar() {
            db.collection("pets").limit(10).get().then(snap => {
                const container = document.getElementById('swipe-container');
                container.innerHTML = "";
                if(snap.empty) container.innerHTML = "Henüz aday yok.";
                snap.forEach(doc => {
                    const d = doc.data();
                    container.innerHTML += `<div class="item-card" style="border-left:5px solid var(--primary)"><div class="item-body"><h3>${d.name} (${d.gender})</h3><p>Aşı: ${d.isVaccinated?'Tam':'Eksik'}</p><button onclick="sendRequest('${doc.id}', '${d.ownerId}')" class="btn-main">Çiftleşme Talebi Gönder</button></div></div>`;
                });
            });
        }

        function sendRequest(pId, oId) {
            const user = auth.currentUser;
            if(!user) return alert("Lütfen önce giriş yapın!");
            db.collection("requests").add({ fromUser: user.uid, toUser: oId, toPet: pId, status: "pending", createdAt: firebase.firestore.FieldValue.serverTimestamp() }).then(() => alert("İstek gönderildi!"));
        }

        function loadRequests() {
            const user = auth.currentUser;
            if(!user) return;
            db.collection("requests").where("toUser", "==", user.uid).onSnapshot(snap => {
                const list = document.getElementById('requests-list');
                list.innerHTML = "";
                snap.forEach(doc => {
                    list.innerHTML += `<div class="item-card" style="border-left:5px solid var(--success)"><div class="item-body"><p>Hayvanınız için çiftleşme talebi var!</p><button class="btn-main" style="background:var(--success)">Kabul Et (Sohbet Başlat)</button></div></div>`;
                });
            });
        }

        function savePet() {
            const user = auth.currentUser;
            if(!user) return alert("Lütfen giriş yapın!");
            db.collection("pets").add({
                name: document.getElementById('pName').value,
                gender: document.getElementById('pGender').value,
                isNeutered: document.getElementById('pNeutered').checked,
                isVaccinated: document.getElementById('pVaccinated').checked,
                ownerId: user.uid,
                createdAt: firebase.firestore.FieldValue.serverTimestamp()
            }).then(() => alert("Patini sisteme kaydettik!"));
        }
